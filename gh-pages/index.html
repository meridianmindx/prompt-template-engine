from prompt_template_engine import Engine

engine = Engine()

# Define your template with variables and guards
template = "You are a {role}. Analyze: {input}"

# Render with validation
result = engine.render(template, role="expert", input="Your text here")

# Or use the safe version with guardrails
safe_result = engine.render_guarded(
    template=template,
    variables={"role": "analyst"},
    guards=["no_injection", "no_leak"],
    max_length=500
)
